kubectl create namespace spring-df

kubectl -n spring-df create -f src/kubernetes/rabbitmq/
kubectl -n spring-df create -f src/kubernetes/kafka/
kubectl -n spring-df create -f src/kubernetes/mysql/

kubectl -n spring-df get all 
kubectl -n spring-df get all -l app=rabbitmq
kubectl -n spring-df get all -l app=kafka
kubectl -n spring-df get all -l app=mysql


kubectl -n spring-df delete all,pvc,secrets -l app=mysql
kubectl -n spring-df delete all -l app=kafka
kubectl -n spring-df delete all -l app=rabbitmq
